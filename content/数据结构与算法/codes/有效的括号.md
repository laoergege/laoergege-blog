```javascript
/**
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    const stack = [];
    const arr = s.split('')
    for(let i = 0; i < arr.length; i++){
        let char = arr[i]
        if(char === '(' || char === '{' || char === '[') {
            stack.push(char)
        } else {
            const top = stack[stack.length - 1]
            if(
                (top === '(' && char === ')') ||
                (top === '{' && char === '}') ||
                (top === '[' && char === ']')
            ){
                stack.pop()
            } else {
                return false
            }
        }
    }
  	// 注意边界条件
    return stack.length ? false : true
};
```
