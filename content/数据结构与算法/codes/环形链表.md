 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */

/**
 * @param {ListNode} head
 * @return {boolean}
 */
var hasCycle = function(head) {
    // 暴力解法，一层循环，hash 缓存循环结果
    // const m = new Map();
    // while(head) {
    //     if(m.get(head)) {
    //         return true
    //     } else {
    //         m.set(head, 1)
    //         head = head.next
    //     }
    // }
    // return false

    // 快慢指针
    let fast = head;
    let slow = head && head.next;
    while(fast && slow){
        fast = fast.next
        slow = slow.next && slow.next.next
        if(fast === slow) {
            return true
        }
    }
    return false
};
