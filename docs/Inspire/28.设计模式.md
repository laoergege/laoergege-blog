- 单例模式：全局只有唯一实例

  ```javascript
  function A(name){
      // 如果已存在对应的实例
     if(typeof A._singleton === 'object'){
         return A._singleton
     }
     //否则正常创建实例
     this.name = name
     
     // 缓存
     A._singleton =this
     return this
  }
  var a1 = new A() 
  var a2= new A()
  console.log(a1 === a2)//true
  ```
